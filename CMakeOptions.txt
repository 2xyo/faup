option(DEBUG "Compile with debug enabled" OFF)
option(LUA "Compile with lua support" OFF)
option(WEBSERVER "Compile with webserver enabled" OFF)

if(${DEBUG})
        set(DEBUG_MODE 1)
        message(STATUS "Compile mode: Debug")
        set(CMAKE_BUILD_TYPE "Debug")
else()
        message(STATUS "Compile mode: Release")
        set(CMAKE_BUILD_TYPE "Release")
endif()


if(${LUA})
        message(STATUS "Lua support: enabled")
else()
        message(STATUS "Lua support: disabled")
	set(FAUP_NOLUA 1)
endif()

if(${WEBSERVER})
        set(WEBSERVER 1)
	add_definitions(-DFAUP_WEBSERVER=1)
        message(STATUS "Built-in Web Server: enabled")
else()
        message(STATUS "Built-in Web Server: disabled")
endif()
